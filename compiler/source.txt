fun mul(var x, var y)
{
	var retval = 0;
	if(x < y)
	{
		while(x > 0)
		{
			retval = retval + y;
			x = x - 1;
		}
	}
	else
	{
		while(y > 0)
		{
			retval = retval + x;
			y = y - 1;
		}
	}
	return retval;
}

fun gradient()
{
	var **g = 512;
	var i = 0;
	while(i < 8)
	{
		*(g + i) = g + mul(i, 16) + 16;
		var j = 0;
		while(j < 8)
		{
			var *addr = *(g + i);
			*(addr + j) = mul(j + 1, i + 1) + 48;
			j = j + 1;
		}
		i = i + 1;

	}
	i = 0;
	
}
